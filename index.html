A Quick Example

Setup

  • npm install -g grunt-cli
  • git init demo
  • cd demo
  • npm init
  • npm install grunt --save-dev
  • mkdir src

Gruntfile


module.exports = function(grunt) {
  grunt.initConfig({
    pkg: grunt.file.readJSON("package.json"),
    ...
  })
}

JSLint

  • npm install grunt-jslint --save-dev

Gruntfile


jslint: {
  files: ["src/**/*.js"],
  directives: {
    devel: true
  },
  options: {
    failOnError: false
  }
}

Watch

  • npm install grunt-contrib-watch --save-dev

Gruntfile


watch: {
  jslint: {
    files: ["src/**/*.js"],
    tasks: ["jslint"],
    options: {
      interrupt: true
    }
  }
}
