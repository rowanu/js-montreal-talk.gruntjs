A Quick Example

Setup

  • npm install -g grunt-cli
  • git init demo
  • cd demo
  • npm init
  • npm install grunt --save-dev

module.exports = (grunt) ->
  grunt.initConfig
    pkg: grunt.file.readJSON('package.json')

CoffeeScript

  • mkdir coffee js
  • npm install grunt-contrib-coffee --save-dev
  • npm install grunt-contrib-watch --save-dev

coffee:
  compile:
    expand: true
    cwd: "coffee/"
    src: ["**/*.coffee"]
    dest: "js/"
    ext: ".js"

Watch

  • npm install grunt-contrib-watch --save-dev

watch:
  coffee:
    files: ["coffee/**/*.coffee"]
    tasks: ["coffee"]
    options:
      interrupt: true
